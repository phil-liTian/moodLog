### 3. 本地存储实现

#### 3.1 存储工具类
```typescript
// utils/storage.ts
class StorageManager {
  private static instance: StorageManager;
  
  static getInstance(): StorageManager {
    if (!this.instance) {
      this.instance = new StorageManager();
    }
    return this.instance;
  }

  // 获取心情记录
  getMoodRecords(startDate?: string, endDate?: string): MoodRecord[] {
    const records = wx.getStorageSync('mood_records') || [];
    if (startDate && endDate) {
      return records.filter((r: MoodRecord) => 
        r.recordDate >= startDate && r.recordDate <= endDate
      );
    }
    return records;
  }

  // 保存心情记录
  saveMoodRecord(record: MoodRecord): void {
    const records = this.getMoodRecords();
    const existingIndex = records.findIndex(r => r.recordId === record.recordId);
    
    if (existingIndex >= 0) {
      records[existingIndex] = record;
    } else {
      records.push(record);
    }
    
    wx.setStorageSync('mood_records', records);
  }

  // 获取今日记录
  getTodayRecord(): MoodRecord | null {
    const today = new Date().toISOString().split('T')[0];
    const records = this.getMoodRecords();
    return records.find(r => r.recordDate === today) || null;
  }
}
```

#### 4.2 心情日历组件
```typescript
// components/mood-calendar/mood-calendar.ts
Component({
  properties: {
    year: {
      type: Number,
      value: new Date().getFullYear()
    },
    month: {
      type: Number,
      value: new Date().getMonth() + 1
    },
    moodData: {
      type: Array,
      value: []
    }
  },
  
  methods: {
    // 生成日历数据
    generateCalendar() {
      const { year, month } = this.data;
      const firstDay = new Date(year, month - 1, 1);
      const lastDay = new Date(year, month, 0);
      const startDate = new Date(firstDay);
      startDate.setDate(startDay.getDate() - firstDay.getDay());
      
      const calendar = [];
      for (let i = 0; i < 42; i++) {
        const date = new Date(startDate);
        date.setDate(startDate.getDate() + i);
        const dateStr = date.toISOString().split('T')[0];
        
        calendar.push({
          date: date.getDate(),
          dateStr,
          mood: this.getMoodForDate(dateStr),
          isCurrentMonth: date.getMonth() === month - 1
        });
      }
      
      this.setData({ calendar });
    },
    
    getMoodForDate(dateStr: string) {
      const moodData = this.data.moodData;
      const dayData = moodData.find(d => d.date === dateStr);
      return dayData ? dayData.mood : null;
    }
  }
});
```

### 6. 数据同步策略

#### 6.1 同步管理器
```typescript
// utils/sync.ts
class SyncManager {
  private static instance: SyncManager;
  
  static getInstance(): SyncManager {
    if (!this.instance) {
      this.instance = new SyncManager();
    }
    return this.instance;
  }
  
  // 执行数据同步
  async syncData(): Promise<void> {
    try {
      // 1. 获取云端数据
      const cloudRecords = await moodAPI.getMoodRecords({
        startDate: '2020-01-01',
        endDate: new Date().toISOString().split('T')[0]
      });
      
      // 2. 获取本地数据
      const localRecords = StorageManager.getInstance().getMoodRecords();
      
      // 3. 数据对比与合并
      const mergedRecords = this.mergeRecords(localRecords, cloudRecords.data.list);
      
      // 4. 上传本地新增记录
      const newRecords = mergedRecords.filter(r => r.recordId.startsWith('local_'));
      for (const record of newRecords) {
        await this.uploadRecord(record);
      }
      
      // 5. 更新本地存储
      StorageManager.getInstance().saveMoodRecords(mergedRecords);
      
    } catch (error) {
      console.error('数据同步失败:', error);
      throw error;
    }
  }
  
  // 合并本地和云端数据
  private mergeRecords(local: MoodRecord[], cloud: MoodRecord[]): MoodRecord[] {
    const merged = [...cloud];
    
    local.forEach(localRecord => {
      const cloudIndex = merged.findIndex(c => c.recordDate === localRecord.recordDate);
      if (cloudIndex >= 0) {
        // 云端存在，以云端为准
        return;
      } else {
        // 本地独有，保留
        merged.push(localRecord);
      }
    });
    
    return merged.sort((a, b) => a.recordDate.localeCompare(b.recordDate));
  }
}
```
